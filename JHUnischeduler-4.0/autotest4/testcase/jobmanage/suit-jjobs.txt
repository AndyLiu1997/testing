case1.用户不指定-u选项时，只能查看自己的作业，当指定-u后，可以查看到其它用户的作业
    [Tags]    p1
    [Setup]    readyTest
    addOrModifyHost    rhela1    2 ()
    addOrModifyHost    rhelb    2 ()
    jadminJhdsSched
    checkClusterStatus
    ${job1}    querySubmitInfo    su user1 -c "jsub sleep 1000"
    ${jobid1}    getJobId    ${job1}
    ${job2}    querySubmitInfo    su user1 -c "jsub sleep 1000"
    ${jobid2}    getJobId    ${job2}
    ${job3}    querySubmitInfo    su user2 -c "jsub sleep 1000"
    ${jobid3}    getJobId    ${job3}
    ${job4}    querySubmitInfo    su user2 -c "jsub sleep 1000"
    ${jobid4}    getJobId    ${job4}
    checkJobStatus    ${jobid1}    RUN
    checkJobStatus    ${jobid2}    RUN
    checkJobStatus    ${jobid3}    RUN
    checkJobStatus    ${jobid4}    RUN
    @{jobids1}    queryUserJobInfo    su user1 -c "jjobs -l"
    ${string1}    Catenate    @{jobids1}
    Should Contain    ${string1}    ${jobid1}
    Should Contain    ${string1}    ${jobid2}
    Should Not Contain    ${string1}    ${jobid3}
    Should Not Contain    ${string1}    ${jobid4}
    @{jobids2}    queryUserJobInfo    su user1 -c "jjobs -u user2 -l"
    ${string2}    Catenate    @{jobids2}
    Should Not Contain    ${string2}    ${jobid1}
    Should Not Contain    ${string2}    ${jobid2}
    Should Contain    ${string2}    ${jobid3}
    Should Contain    ${string2}    ${jobid4}
    @{jobids3}    queryUserJobInfo    su user1 -c "jjobs -u 'user1 user2' -l"
    ${string3}    Catenate    @{jobids3}
    Should Contain    ${string3}    ${jobid1}
    Should Contain    ${string3}    ${jobid2}
    Should Contain    ${string3}    ${jobid3}
    Should Contain    ${string3}    ${jobid4}
    [Teardown]    modHostInfo    rhela1=1 ()    rhelb=1 ()

case2.jjobs -a查看所有状态的作业
    [Tags]    p1
    [Setup]    readyTest
    addOrModifyParam    CLEAN_PERIOD    5
    jadminJhdsSched
    checkClusterStatus
    sleep    10
    addOrModifyParam    CLEAN_PERIOD    172800
    jadminJhdsSched
    checkClusterStatus
    ${job1}    querySubmitInfo    su user1 -c "jsub sleep 1"
    ${jobid1}    getJobId    ${job1}
    ${job2}    querySubmitInfo    su user1 -c "jsub sleep 1000"
    ${jobid2}    getJobId    ${job2}
    checkJobStatus    ${jobid2}    RUN
    runCommand    jctrl kill ${jobid2}
    checkJobStatus    ${jobid2}    EXIT
    checkJobStatus    ${jobid1}    DONE
    ${job3}    querySubmitInfo    su user1 -c "jsub -q q1 sleep 1000"
    ${jobid3}    getJobId    ${job3}
    ${job4}    querySubmitInfo    su user1 -c "jsub sleep 1000"
    ${jobid4}    getJobId    ${job4}
    checkJobStatus    ${jobid3}    RUN
    checkJobStatus    ${jobid4}    RUN
    ${job5}    querySubmitInfo    su user1 -c "jsub \ -q q3 sleep 1000"
    ${jobid5}    getJobId    ${job5}
    ${job6}    querySubmitInfo    su user1 -c "jsub sleep 1000"
    ${jobid6}    getJobId    ${job6}
    ${job7}    querySubmitInfo    su user1 -c "jsub sleep 1000"
    ${jobid7}    getJobId    ${job7}
    ${job8}    querySubmitInfo    su user1 -c "jsub -J arr[1-2] sleep 1000"
    ${jobid8}    getJobId    ${job8}
    runCommand    su user1 -c "jctrl stop ${jobid4} ${jobid6}"
    checkJobStatus    ${jobid1}    DONE
    checkJobStatus    ${jobid2}    EXIT
    checkJobStatus    ${jobid3}    SSUSP
    checkJobStatus    ${jobid4}    USUSP
    checkJobStatus    ${jobid5}    RUN
    checkJobStatus    ${jobid6}    PSUSP
    checkJobStatus    ${jobid7}    PEND
    checkJobStatus    ${jobid8}[1]    PEND
    checkJobStatus    ${jobid8}[2]    PEND
    @{jobids3}    queryUserJobInfo    su user1 -c "jjobs -a -l"
    ${string3}    Catenate    @{jobids3}
    Should Contain    ${string3}    ${jobid1}
    Should Contain    ${string3}    ${jobid2}
    Should Contain    ${string3}    ${jobid3}
    Should Contain    ${string3}    ${jobid4}
    Should Contain    ${string3}    ${jobid5}
    Should Contain    ${string3}    ${jobid6}
    Should Contain    ${string3}    ${jobid7}
    Should Contain    ${string3}    ${jobid8}[1-2]
    [Teardown]    killJob
